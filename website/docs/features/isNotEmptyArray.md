# isNotEmptyArray

Check if the array is not empty.

## Usage

```tsx
import { isNotEmptyArray } from "lulinjs";

const result = isNotEmptyArray([1]);
// Output: true
```

```jsx live
import {isNotEmptyArray} from "./lulin";

function IsNotEmptyArray() {
    return (
        <div>
            <h4>Result:</h4>
            <h2 style={{color: 'rgb(102, 212, 189)'}}>
                {isNotEmptyArray([1])}
            </h2>
        </div>
    );
}
```
